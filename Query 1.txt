Query 1:SELECT ar.name artist, MIN (i.unitprice * i.quantity) minimum_earnings FROM artist ar JOIN album al ON ar.artistid = al.artistid JOIN track t ON al.albumid = t.albumid JOIN invoiceline i ON t.trackid = i.trackid GROUP BY ar.artistid ORDER BY minimum_earnings DESC LIMIT 5;Query 2:SELECT ar.name artist, Sum(i.unitprice * i.quantity) total_earnings FROM artist ar JOIN album al ON ar.artistid = al.artistid JOIN track t ON al.albumid = t.albumid JOIN invoiceline i ON t.trackid = i.trackid GROUP BY ar.artistid ORDER BY total_earnings DESC LIMIT 10;Query 3:SELECT MIN (total)FROM InvoiceJOIN CustomerON invoice.customerid = customer.customeridGROUP BY invoice.totalORDER BY invoice.totalASC LIMIT 10;Query 4:SELECT max (total)FROM InvoiceJOIN CustomerON invoice.customerid = customer.customeridGROUP BY invoice.totalORDER BY invoice.totalDESC LIMIT 15;
